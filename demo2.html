  <h1 id="title">GIS Web Demo</h1>
  <button id="zoomBtn">Zoom to Feature</button>
  <div id="map"></div>
  <script src="https://unpkg.com/leaflet/dist/leaflet.js"></script>
  <script>
    // Variables
    const mapCenter = [51.05, -114.07]; // Downtown Calgary
    let zoomLevel = 13;

    // Initialize map
    const map = L.map('map').setView(mapCenter, zoomLevel);

    // Add OpenStreetMap tile layer
    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
      attribution: '© OpenStreetMap contributors'
    }).addTo(map);

    // Add a sample feature to zoom to
    const sampleFeature = L.circle(mapCenter, {
      radius: 400,
      color: 'green'
    }).addTo(map);

    // Add a marker with popup
    L.marker([51.05, -114.07])
      .addTo(map)
      .bindPopup('Downtown Calgary')
      .openPopup();

    // Function
    function zoomToFeature() {
      map.fitBounds(sampleFeature.getBounds());
    }

    // DOM Manipulation
    const title = document.getElementById("title");
    title.style.color = "darkgreen";

    // Event Listener
    document.getElementById("zoomBtn").addEventListener("click", zoomToFeature);
  </script>
